**`async def create(thread_ts: str, description: str) -> str`**

**`def is_code_task(prompt: str) -> bool`**

lower = prompt.lower()
return not any(lower.startswith(kw) or f" {kw} " in lower for kw in readonly_keywords)

Default to `True` (code task) when uncertain — worktrees are cheap.

import ast, sys

def check_param(filepath, funcname, param):
    with open(filepath) as f: src = f.read()
    tree = ast.parse(src)
    for node in ast.walk(tree):
        if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)) and node.name == funcname:
            args = [a.arg for a in node.args.args] + [a.arg for a in node.args.kwonlyargs]
            defaults_with_none = [a.arg for a, d in zip(reversed(node.args.args), reversed(node.args.defaults)) if isinstance(d, ast.Constant) and d.value is None]
            if param not in args + defaults_with_none:
                # check with default None via defaults
                all_args = [a.arg for a in node.args.args]
                if param not in all_args:
                    sys.exit(f'{filepath}:{funcname} missing param {param}')
            return
    sys.exit(f'{filepath}: function {funcname} not found')

check_param('bot/agent.py', 'run_agent', 'cwd')
